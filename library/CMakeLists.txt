cmake_minimum_required(VERSION 3.15)
project(hyperscan_cpp_library)
set(CMAKE_CXX_STANDARD 14)
#add_link_options("-static-libgcc")
add_link_options("-static-libstdc++")
find_package(PkgConfig REQUIRED)
set(ENV{PKG_CONFIG_PATH} "${CMAKE_CURRENT_SOURCE_DIR}/../hyperscan/build/install/lib/pkgconfig")
pkg_check_modules(HPS REQUIRED libhs)
add_library(hsc hyperscan_cpp.h hyperscan_cpp.cpp)
target_link_directories(hsc PUBLIC ${HPS_LIBRARY_DIRS})
target_link_libraries(hsc PUBLIC ${HPS_LIBRARIES})
target_include_directories(hsc PUBLIC ${HPS_INCLUDE_DIRS} ${CMAKE_CURRENT_SOURCE_DIR})
